	by default class->attr change will change
	the rendered but not the other way around

	to bind one,
		<form>
			<input type="text"
				wire:model="<propertyName>"
			>
		</form>

	so changes in the class will reflect the input
	and changes in the input value will also reflect
	in the class

	1
		refresh its current property values by
		<form>
			<input type="text"
				wire:model="<propertyName>"
			>
			<button
				// Refreshes the variables' values
				// .prevents - Prevents refreshing the browser
				wire:click.prevent="$refresh"
			>Update
			</button>
		</form>

	2
		no more .prevent. Laravel does it automatically

		<form wire:submit="$refresh">
			<input type="text"
				wire:model="<propertyName>"
			>
			<button>Update</button>
		</form>


	3 real-time
		adding .live makes it live

		<form wire:submit="$refresh">
			<input type="text"
				wire:model.live="<propertyName>"
			>
			<button>Update</button>
		</form>

		.debounce - checks if you stopped typing then refresh
		.debounce.000ms

	class {
		public string $title;
		public string $author;
		public int $rating;

		public function save() {
			Book::create(
				"title" => $this->title,
				"author" => $this->author,
				"rating" => $this->rating,
			);

			$this->redirect('/')
		}
	}


Validation Rules
	1
		class {
			...

			public function save() {
				$this->validate([
					"title" => "string|requried|min:3|max:50",
					"author" => "string|requried|min:3|max:50",
					"rating" => "integer|requried|min:1|max:10",
				]);

				Book::create(...);

				$this->redirect('/')
			}
		}

	2
		class {
			#[Rule("string|requried|min:3|max:50")]
			public string $title;

			#[Rule("string|requried|min:3|max:50")]
			public string $author;

			#[Rule("integer|requried|min:1|max:10")]
			public int $rating;

			public function save() {
				$this->validate();

				Book::create(...);

				$this->redirect('/')
			}
		}
