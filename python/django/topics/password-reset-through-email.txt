
PASSWORD RESET THROUGH EMAIL LINK AND DEFAULT DJANGO VIEWS
	# This method sends the user an email with a link to reset their password.
	# The link includes a secure token
	# Django's default views will handled validations automatically and all we
	# 	have to do just import the view, route it, and create the html structure ourselves

	CREATING THE TEMPLATES
		on PROJECT_FOLDER/APP_NAME/templates/password_reset.html 
			# This is the page to collect the user's email

			...
		    <form method="POST">
		        {% csrf_token %}
    			{{ form }}
    			<button type="submit">SEND EMAIL</button>
		    </form>
			...


		on PROJECT_FOLDER/APP_NAME/templates/password_reset_done.html 
			# This page shows that the email has been sent

			...
			    <div>AN EMAIL HAS BEEN SENT</div>
			    <a href="LOGIN_URL">LOGIN</a>
			...


		on PROJECT_FOLDER/APP_NAME/templates/password_reset_confirm.html 
			# This page requests for the user's new password

		    <form method="post">
		        {% csrf_token %}
        		{{ form }}
    			<button type="submit">CHANGE PASSWORD</button>
		    </form>

	ROUTING
		on PROJECT_NAME/APP_NAME/urls.py
			from django.contrib.auth.views import (
				PasswordResetView, # The view to get the user's email
				PasswordResetDoneView, # The page to show that the email has been sent
				PasswordResetConfirmView, # The page where thte user creates a new password
				PasswordResetCompleteView # The page to show that the password has been changed
			)

			urlpatterns = [
				...
				path('URL/', PasswordResetView.as_view(
					template_name='APP_NAME/password_reset.html'), 
					name='password-reset'),

				path('URL_1/', PasswordResetDoneView.as_view(
					template_name='APP_NAME/password_reset_done.html'), 
					name='password_reset_done'),

				path('URL_2/<uidb64>/<token>/', PasswordResetConfirmView.as_view(
					template_name='APP_NAME/password_reset_confirm.html'), 
					name='password_reset_confirm'),

				path('URL_3/', PasswordResetCompleteView.as_view(
					template_name='APP_NAME/password_reset_complete.html'), 
					name='password_reset_complete'),
			]

	SETUP EMAILING ACCOUNT
		GENERATING AN APP PASSWORD
			> Go to your email domain provider (E.g.: Google & Yahoo) look for App Password
			> Generate the app password (The name doesnt matter)
			> Note down the email and the password generated

		SETUP AN ENVIRONMENT VARIABLE
			# This makes sure that your app password wont be public (E.g.: Uploading the code in GitHub)

			OPTION 1: Windows
				> Search for "Environment Variables"
				> Create two variables:
					1. EMAIL_USER with the email as the value
					2. EMAIL_PASS with the generated password as the value

			OPTION 2: Linux
				# This is the method I personally use because I find this easier
				# This method should work in Windows option as well.

				CREATING AN ENVIRONMENT VARIABLE
					on PROJECT_FOLDER/.env
						EMAIL_USER="YOUR_EMAIL"
						EMAIL_PASS="GENERATED_PASSWORD"

				EXCLUDING ENVIRONMENT FILE (.env) IN PUSH
					on PROJECT_FOLDER/.gitignore
						.env

				> Install python-dotenv
					>>> pip install python-dotenv

				on PROJECT_FOLDER/PROJECT_NAME/settings.py
					...
					from dotenv import load_dotenv
					load_dotenv() # Loads the environment variables
					...
