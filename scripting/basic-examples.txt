
#!/bin/bash

You can use any terminal command in bash script 
	ls
	mv
	cp
	mkdir
	zip
	sudo apt update
	curl
	ping
	grep
	find

create new file
	>>> nano hello.sh

making things exec
	>>> chmod +x fileName.sh

running
	>>> ./fileName.sh

VARIABLES
	name=value // no spaces between =
	echo "Hello, $name" // echo is print

COMMENT
	# This is a comment

EXIT CODES
	echo "Something failed"
	exit 1
	# exit 0 means success, anything else means error.

ARGUMENTS 
	>>> ./myscript.sh foo bar baz

	Inside the script:
		$0 → script name (myscript.sh)
		$1 → "foo"
		$2 → "bar"
		$3 → "baz"
		$@ → all args as separate words
		$* → all args as one string
		$# → number of arguments passed

	Example:
		#!/bin/bash

		echo "Script name: $0"
		echo "First arg: $1"
		echo "Total args: $#"

		for arg in "$@"; do
		  echo "Arg: $arg"
		done

	the "--" and "-" in the args we see are just conventions
		"-n" for shorthands
		"--paramName" for full context




String Tests
	Expression	Meaning
	-z "$var"	String is empty (zero length)
	-n "$var"	String is not empty
	"$a" = "$b"	Strings are equal
	"$a" != "$b"	Strings are not equal


Number Tests
	Expression	Meaning
	-eq	equal to
	-ne	not equal
	-lt	less than
	-le	less than or equal
	-gt	greater than
	-ge	greater than or equal


Loops
	# For loop (fixed list)
	for item in a b c; do
	  echo "$item"
	done

	# While loop
	while [ condition ]; do
	  ...
	done

Conditionals
	if [ "$x" = "yes" ]; then
	  echo "Yes"
	elif [ "$x" = "no" ]; then
	  echo "No"
	else
	  echo "Other"
	fi

Functions
	greet() {
	  echo "Hello, $1"
	}

	greet "Eya"  # → Hello, Eya


Arguments & Shifting
	Symbol	Meaning
	$0	Script name
	$1	First argument
	$2	Second argument
	$@	All arguments (individually)
	$*	All arguments (as one string)
	$#	Number of arguments
	shift	Discard $1, move next up

 Argument Parsing Pattern
	while [ "$1" ]; do
	  case "$1" in
	    --from)
	      from="$2"
	      shift 2
	      ;;
	    --dry-run)
	      dryrun=true
	      shift
	      ;;
	    --help|-h)
	      echo "Help text"; exit 0
	      ;;
	    *)
	      echo "Unknown option: $1"; exit 1
	      ;;
	  esac
	done

Useful Commands in Scripts
	Command	Use
	ls, cp, mv, rm	File operations
	mkdir -p	Make directory (no error if exists)
	read	Read user input
	echo, printf	Output
	date, cut, grep, sed, awk	Text tools
