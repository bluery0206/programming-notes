ARGUMENTS 
	>>> ./myscript.sh foo bar baz

	Inside the script:
		$0 → script name (myscript.sh)
		$1 → "foo"
		$2 → "bar"
		$3 → "baz"
		$@ → all args as separate words
		$* → all args as one string
		$# → number of arguments passed

	Example:
		#!/bin/bash

		echo "Script name: $0"
		echo "First arg: $1"
		echo "Total args: $#"

		for arg in "$@"; do
		  echo "Arg: $arg"
		done

	the "--" and "-" in the args we see are just conventions
		"-n" for shorthands
		"--paramName" for full context


FILE CHECKS 
	if [ -f "file.txt" ]; then...
	if [ -d "/etc" ]; then...
	if [ -e "something" ]; then...

	Other flags:
		-s → exists and not empty
		-r / -w / -x → readable / writable / executable
		! -f → not a file



SAMPLE MULTI PARAMETER
	#!/bin/bash

	# WHILE there are arguments left
	while [ $# -gt 0 ]; do

		# LOOK at the current argument
		case "$1" in
			# IF it matches "--from":
			--from)
				# STORE the next argument as `from_file`
				from="$2"

				# SKIP the current and next arguments (shift 2)
				shift 2
				;;
			--to)
				to="$2"
				shift 2
				;;
			--dry-run)
				dryrun=true
				shift
				;;
			--help|-h)
				echo "Usage: $0 --from FILE --to FILE [--dry-run]"
				exit 0
				;;
			*)
				echo "Unknown option: $1"
        		# EXIT with error
				exit 1
				;;
		esac
	done

	# Do something with collected values
	echo "FROM: $from"
	echo "TO: $to"
	echo "Dry run: $dryrun"




File & Directory Tests
	Expression	Meaning
	-e FILE	File exists
	-f FILE	File exists and is a regular file
	-d DIR	Directory exists
	-s FILE	File exists and is not empty
	-r FILE	File is readable
	-w FILE	File is writable
	-x FILE	File is executable
	! -e FILE	File does not exist


String Tests
	Expression	Meaning
	-z "$var"	String is empty (zero length)
	-n "$var"	String is not empty
	"$a" = "$b"	Strings are equal
	"$a" != "$b"	Strings are not equal


Number Tests
	Expression	Meaning
	-eq	equal to
	-ne	not equal
	-lt	less than
	-le	less than or equal
	-gt	greater than
	-ge	greater than or equal

Arguments & Shifting
	Symbol	Meaning
	$0	Script name
	$1	First argument
	$2	Second argument
	$@	All arguments (individually)
	$*	All arguments (as one string)
	$#	Number of arguments
	shift	Discard $1, move next up

 Argument Parsing Pattern
	while [ "$1" ]; do
	  case "$1" in
	    --from)
	      from="$2"
	      shift 2
	      ;;
	    --dry-run)
	      dryrun=true
	      shift
	      ;;
	    --help|-h)
	      echo "Help text"; exit 0
	      ;;
	    *)
	      echo "Unknown option: $1"; exit 1
	      ;;
	  esac
	done
