OPERATORS
	ARITHMETIC
		x + y // Adds together two values
		x - y // Subtracts one value from another
		x * y // Multiplies two values
		x / y // Divides one value by another
		x % y // Returns the division remainder
		++x // Increases the value of a variable by 1
		--x // Decreases the value of a variable by 1

	ASSIGNMENT
		Operator	Example		Same As
		=			x = 5		x = 5	
		+=			x += 3		x = x + 3	
		-=			x -= 3		x = x - 3	
		*=			x *= 3		x = x * 3	
		/=			x /= 3		x = x / 3	
		%=			x %= 3		x = x % 3	
		&=			x &= 3		x = x & 3	
		|=			x |= 3		x = x | 3	
		^=			x ^= 3		x = x ^ 3	
		>>=			x >>= 3		x = x >> 3	
		<<=			x <<= 3		x = x << 3

	COMPARISON
		x == y 	Equal to
		x != y 	Not equal
		x > y 	Greater than	
		x < y 	Less than
		x >= y 	Greater than or equal to
		x <= y 	Less than or equal to

	LOGICAL
		x < 5 &&  x < 10 	Returns true if both statements are true		
		x < 5 || x < 4 		Returns true if one of the statements is true		
		!					Reverse the result, returns false if the result is true
