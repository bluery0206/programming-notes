	> Login and Log out
		on Http/Controllers/ContollerName.php
			class ContollerName extends Controller {
				public function methodName(Request $request) {

					# Validating fields
						user Illuminate\Validation\Rule;

						$fields = $request->validation([
							'field_one' => ['required', 'min:3', Rule::unique('users', 'name')],
							'field_two' => 'required',
							'email' => ['required', 'email', Rule::unique('users', 'email')],
							...
							'field_n' => 'required',
					])

					# hashing password
					$fields['password'] = bcrypt($fields['passworcd']);

					# Model is the metaphor for table/data structures
					$user = User::create($fields)

					auth()->login($user);

					return redirect('/');
				}	
			}
