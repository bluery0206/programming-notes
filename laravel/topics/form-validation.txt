CONTROLLER
	public function store(Request $request) {
		$validated = $request->validate([
			'title' => 'required|max:255',
			'content' => 'required|min:10',
		]);

		Task::create($validated);

		return redirect()->route("home")->with('success', 'Add Successful');
	}
