				rename	: ReyCastRight
				face	: Right
			RayCast2D
				rename	: ReyCastLeft
				face	: Left

	# Script
		var SPEED 		= n
		var DIRECTION 	= 1		# 1=right, -1=left

		@onready var ray_cast_right = $ReyCastRight
		@onready var ray_cast_left	= $ReyCastLeft
		@onready var animated_spr	= $AnimatedSprite2D


		func _process(delta):

			# Changing Move Direction
			if ray_cast_right.is_colliding():
				DIRECTION		= -1
				animated_spr	= true
			if ray_cast_left.is_colliding():
				DIRECTION		= 1
				animated_spr	= false

			# Move
			position.x += DIRECTION * SPEED * delta
